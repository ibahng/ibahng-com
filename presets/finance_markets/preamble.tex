\newcommand{\upslope}[1]{
  \def\linepos{#1}
  \addplot[domain=0.5:9.5, samples=100, thick, restrict y to domain=0.5:9.5] {x + \linepos}; 
}

\newcommand{\downslope}[1]{
  \def\linepos{#1}
  \addplot[domain=0.5:9.5, samples=100, thick, restrict y to domain=0.5:9.5] {-x + 10 + \linepos}; 
}

\newcommand{\verticalslope}[1]{
  \draw[thick] (axis cs:#1,0.5) -- (axis cs:#1,9.5);
}

\newcommand{\horizontalslope}[1]{
  \draw[thick] (axis cs:0.5,#1) -- (axis cs:9.5,#1);
}

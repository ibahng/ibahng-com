\usepackage{float} % for better figure placement using [h]
\usepackage{graphicx} % for importing figures into the tex file
\usepackage{subcaption} % allows for multiple subfigures with individual captions wihtin one figure
\usepackage{xparse} % allows for more than 9 parameters in commands

\usepackage{silence}

\newcommand{\myfig}[4]{%
  \begin{figure}[H]
    \centering
    \adjustbox{valign=c}{\includegraphics[width=#1\textwidth]{#2}}
    \caption{#3}
    \label{#4}
  \end{figure}
}

\newcommand{\sbs}[3]{
  \begin{minipage}[t]{#1\textwidth}
    \setlength{\parskip}{0.8em}
    #2
    \vspace{4pt}
  \end{minipage}
  \hfill
  \begin{minipage}[t]{\textwidth-#1\textwidth-1em}
    \vspace{-1em}
    #3
  \end{minipage}
}

\newcommand{\mymultifig}[8]{

    \centering%
    \begin{subfigure}{#1\textwidth}%
      \centering%
      \includegraphics[width=\linewidth]{#2}%
    \end{subfigure}%
    \hfill%
    \begin{subfigure}{#3\textwidth}%
      \centering%
      \includegraphics[width=\linewidth]{#4}%
    \end{subfigure}%
    \hfill%
    \begin{subfigure}{#5\textwidth}%
      \centering%
      \includegraphics[width=\linewidth]{#6}%
    \end{subfigure}%

    \label{#8}%
  \end{figure}%
}

\newcommand{\pgfig}[4]{%

      \end{tikzpicture}
    }
    \caption{#2}
    \label{#3}
  \end{figure}
}

\newcommand{\doublepgfig}[5]{%
  \begin{figure}[H]
    \centering
    \scalebox{#5}{
      \begin{tikzpicture}
        #1
      \end{tikzpicture}
      \qquad
      \qquad
      \begin{tikzpicture}
        #2
      \end{tikzpicture}
    }
    \caption{#3}
    \label{#4}
  \end{figure}
}
